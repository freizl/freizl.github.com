### "look around"
$test = "see Jeffs book";
$test =~ s/(?<=Jeff)(?=s)/'/g;
print "$test \n";    ### expect see Jeff's book

### Start of match ( or end of previous match): \G
$test = "abcde";
$test =~ s/\Gx?/!/g;
print "$test \n";             ### expect !abcde
$test =~ s/\x?/!/g;
print "$test \n";             ### expect !a!b!c!d!e
