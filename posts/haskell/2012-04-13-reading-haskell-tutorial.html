instance (Eq a) => Eq (Tree a) where
  (Leaf x) == (Leaf y)  = x == y
  (Branch l r) == (Branch l' r') = l == l' && r == r'
  _ == _ = False

data Tree a = Leaf a | Branch (Tree a) (Tree a) deriving Eq
