> module Main where
> 
> sumWith1 v [] = v
> sumWith1 v (x:xs) = sumWith1 (v+x) xs
> 
> sumWith2 v [] = v
> sumWith2 v (x:xs) = (sumWith2 $! (v+x)) xs
> 
> test = sumWith2 0 [1..20000000]
> 
> main = print test
